country_event = {
	id = uzb_sart.1
	title = uzb_sart.1.t
	desc = uzb_sart.1.d
	icon = "gfx/interface/icons/event_icons/tensions.dds"
	is_triggered_only = yes
	cooldown = { years = 3 }

	trigger = {
		c:UZB ?= THIS
		has_journal_entry = je_uzbekistan_identity
		NOT = { has_variable = uzb_identity_resolved }
		check_variable = { name = identity_timer value > 0 }
	}

	option = {
		name = uzb_sart.1.a
		default_option = yes
		change_variable = { name = uzb_identity_uzbek_progress add = 10 }
		add_radicals = { culture = cu:tajik value = 0.20 }
	}
	option = {
		name = uzb_sart.1.b
		change_variable = { name = uzb_identity_tajik_progress add = 10 }
		add_radicals = { culture = cu:uzbek value = 0.20 }
	}
	option = {
		name = uzb_sart.1.c
		change_variable = { name = uzb_identity_middle_progress add = 5 }
		add_modifier = { name = uzb_modifier_bazaar_mediation months = 36 }
		add_loyalists = { culture = cu:uzbek value = 0.05 }
		add_loyalists = { culture = cu:tajik value = 0.05 }
	}
}

country_event = {
	id = uzb_sart.10
	title = uzb_sart.10.t
	desc = uzb_sart.10.d
	icon = "gfx/interface/icons/event_icons/victory.dds"
	is_triggered_only = yes

	trigger = {
		c:UZB ?= THIS
		NOT = { has_variable = uzb_identity_resolved }
	}

	immediate = {
		set_variable = uzb_identity_resolved
	}

	option = {
		name = uzb_sart.10.a
		default_option = yes
		add_modifier = { name = uzb_modifier_uzbek_primacy days = -1 }
		add_primary_culture = cu:uzbek
		remove_primary_culture = cu:tajik
		add_radicals = { culture = cu:tajik value = 0.30 }
		if = {
			limit = { has_technology_researched = nationalism }
			start_secession = { culture = cu:tajik state = s:STATE_SAMARQAND }
			start_secession = { culture = cu:tajik state = s:STATE_BUKHARA }
		}
	}
}

country_event = {
	id = uzb_sart.40
	title = uzb_sart.40.t
	desc = uzb_sart.40.d
	icon = "gfx/interface/icons/event_icons/chaos.dds"
	is_triggered_only = yes

	trigger = {
		c:UZB ?= THIS
		NOT = { has_variable = uzb_identity_resolved }
	}

	immediate = {
		set_variable = uzb_identity_resolved
	}

	option = {
		name = uzb_sart.40.a
		default_option = yes
		add_modifier = { name = uzb_modifier_identity_crisis days = 3650 }
		add_radicals = { culture = cu:uzbek value = 0.30 }
		add_radicals = { culture = cu:tajik value = 0.30 }
		if = {
			limit = { has_technology_researched = nationalism }
			if = {
				limit = {
					check_variable = { name = uzb_identity_tajik_progress value >= var:uzb_identity_uzbek_progress }
				}
				start_secession = { culture = cu:tajik state = s:STATE_SAMARQAND }
				start_secession = { culture = cu:tajik state = s:STATE_BUKHARA }
			}
			else = {
				start_secession = { culture = cu:uzbek state = s:STATE_TOSHKENT }
				start_secession = { culture = cu:uzbek state = s:STATE_FERGANA }
				start_secession = { culture = cu:uzbek state = s:STATE_KHIVA }
			}
		}
	}
}
